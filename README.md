import SwiftUI


struct ContentView: View {
    
@State var splashScreen  = true

var body: some View {
     ZStack{
        Group{
          if splashScreen {
                SplashScreen()
           }
          else{
              MainView()
                }
            }
           .onAppear {
              DispatchQueue
                   .main
                   .asyncAfter(deadline:
                    .now() + 4) {
               self.splashScreen = false
              }
            }
        }
    }
}
